class FeatureStore:
    def _init_(self):
        self.features = {}

    def add_feature(self, feature_name, data):
        if feature_name not in self.features:
            self.features[feature_name] = {'data': data, 'version': 1}
        else:
            self.features[feature_name]['data'] = data
            self.features[feature_name]['version'] += 1

    def get_feature(self, feature_name):
        if feature_name in self.features:
            return self.features[feature_name]['data']
        else:
            print(f"Feature '{feature_name}' not found in the Feature Store.")
            return None

    def get_feature_version(self, feature_name):
        if feature_name in self.features:
            return self.features[feature_name]['version']
        else:
            print(f"Feature '{feature_name}' not found in the Feature Store.")
            return None

		feature_store.add_feature('age', [25, 30, 35, 40, 45])

		age_feature = feature_store.get_feature('age')

		# Add a new version of the feature
		feature_store.add_feature('age', [26, 31, 36, 41, 46])

		# Get the updated feature and version
		updated_age_feature = feature_store.get_feature('age')
		version = feature_store.get_feature_version('age')

		print(f"Updated Age Feature: {updated_age_feature}")
		print(f"Feature Version: {version}")
